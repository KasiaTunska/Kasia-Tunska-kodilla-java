SELECT ISSUES.SUMMARY, I.NAME
FROM ISSUESLISTS I
         JOIN ISSUES ON I.ID = ISSUES.ISSUESLIST_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
         JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

SELECT  U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_TASKS
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.FIRSTNAME, U.LASTNAME
HAVING COUNT(*) >1;
